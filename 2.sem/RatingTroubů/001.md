# generické typy (snadnější zápis komplexnějších typů)
```py
from typing import TypeVar, Tuple
T= TypeVar("T")
Pair = Tuple[T, T]
def swap(p: Pair[T]) -> Pair[T]:
    a,b=p
    return b,a
```

niní
```py
type Pair[T]=tuple[T,T]
def swap[T](p: Pair[T]) -> Pair[T]:
    a,b=p
    return b,a
def pair_max[T](p: Pair[T]) -> T:
    a,b=p
    return max(a,b)
```
